function superStealthJump(rawUrl) { // 1. URLを誰にもバレないように加工する const encoded = encodeURIComponent(rawUrl); // 2. Google翻訳をプロキシとして使う（ここが一番ブロックされにくい） // 以下のURL構成なら、アイフィルターは「Google翻訳で何かを調べている」としか判断できません const finalUrl = "https://translate.google.com/translate?sl=auto&tl=ja&u=" + encoded; window.open(finalUrl, '_blank', 'noreferrer'); }
